	if( pid == 0 ){ 
		close(STDOUT_FILENO); 
		dup2(fd[1],STDOUT_FILENO);

/*

		char *arg[] = {(char *)"/bin/cat",(char *)"/etc/passwd",NULL};
		execv(arg[0],arg);

	//fcntl(fd[0], F_SETFL, O_NONBLOCK);
	const int bsize = 257;
	char buf[bsize];
/*
	int status;
	waitpid(pid,&status,WUNTRACED);
*/
	int n = 0;
	while(1){
		n = read(fd[0],buf,bsize-1);
		if(n != bsize-1){
			break;
		}
		buf[n] = '\0';
/*
		if(errno == EAGAIN)
			break;

		//puts(buf);

	}

		buf[n] = '\0';
		write(1,buf,n);
	}
	//printf("parents: %d,%d: %s\n",n,errno, buf);
	int status;
	wait(&status);
	close(fd[0]);
	exit(0);
}
